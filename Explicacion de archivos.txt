services: 						# Define el grupo de contenedores que conforman la aplicación
  backend_node: 					# Nombre del servicio para la aplicación Node.js
    container_name: backend_node 			# Nombre que tendrá el contenedor al ejecutarse
    build: 						# Instrucciones para construir la imagen personalizada
      context: ./ 					# Ubicación de los archivos del proyecto (carpeta actual)
      dockerfile: Dockerfile 				# Nombre del archivo con las reglas de construcción
    ports: 						# Configuración de comunicación con el exterior
      - "81:3000" 					# Mapea el puerto 81 de tu PC al 3000 interno del contenedor
    environment: 					# Variables de entorno para configurar la app
      PORT: ${PORT} 					# Toma el valor del puerto desde un archivo .env externo
    depends_on: 					# Establece el orden de encendido
      - base_datos_mysql 				# Indica que Node no arrancará hasta que la DB esté lista
    restart: unless-stopped 				# Se reinicia solo si falla, a menos que tú lo detengas

  base_datos_mysql: 					# Nombre del servicio de la base de datos
    container_name: base_datos_mysql 			# Nombre del contenedor para la base de datos
    image: mysql:latest 				# Descarga la versión más reciente de MySQL oficial
    environment: 					# Configuración inicial de seguridad y datos
      MYSQL_ROOT_PASSWORD: root 			# Define la contraseña maestra del administrador
      MYSQL_DATABASE: bd_base_defecto 			# Crea automáticamente esta base de datos al iniciar
      MYSQL_PASSWORD: userpassword 			# Define la contraseña para un usuario normal
    ports: 						# Puertos para la base de datos
      - "3306:3306" 					# Puerto estándar de MySQL abierto hacia tu computadora
    volumes: 						# Persistencia de datos
      - mysql_data_service:/var/lib/mysql 		# Guarda los datos en un volumen para no perderlos al borrar el contenedor
    restart: unless-stopped 				# Mantiene la base de datos siempre encendida

  db_phpmyadmin: 					# Servicio para la interfaz gráfica de la base de datos
    container_name: db_phpmyadmin 			# Nombre del contenedor de la herramienta web
    image: phpmyadmin/phpmyadmin 			# Usa la imagen oficial de phpMyAdmin
    environment: 					# Configuración de conexión automática
      PMA_HOST: base_datos_mysql 			# Le dice a phpMyAdmin a qué servidor conectarse
      PMA_PORT: 3306 					# Puerto de comunicación interna entre contenedores
      PMA_USER: root 					# Usuario para el inicio de sesión automático
      PMA_PASSWORD: root 				# Contraseña para el inicio de sesión automático
    ports: 						# Puerto para ver la interfaz en el navegador
      - "7777:80" 					# Entrarás a ver tus tablas escribiendo localhost:7777
    depends_on: 					# Dependencia lógica
      - base_datos_mysql 				# No inicia sin la base de datos presente
    restart: unless-stopped 				# Siempre activo para gestión rápida

volumes: 						# Sección para definir el almacenamiento persistente
  mysql_data_service: 					# Nombre del volumen que creaste arriba para MySQL

########## producto.controller.js


const Producto = require("./../models/Producto");                                // Importa el modelo para realizar operaciones en la BD

async function listar(req, res){                                                 // Función para obtener todos los registros
    try {                                                                        // Inicia bloque de seguridad para capturar errores
        const productos = await Producto.findAll();                              // Ejecuta: SELECT * FROM productos (promesa asíncrona)
        return res.status(200).json(productos);                                  // Responde con éxito (200) enviando la lista de productos
    } catch (error) {                                                            // Captura fallos si la base de datos no responde
        console.log(error);                                                      // Muestra el error técnico en la consola del servidor
        return res.status(500).json({mensaje: "Error al obtener los productos"}) // Responde con error de servidor (500) al cliente
    }
}

async function guardar(req, res){                                                // Función para crear un nuevo registro
    try {                                                                        // Inicia bloque de seguridad
        await Producto.create(req.body);                                         // Toma los datos del formulario y los inserta en la BD
        return res.status(201).json({mensaje: "Producto registrado"})            // Responde con éxito de creación (201)
    } catch (error) {                                                            // Captura fallos de validación o conexión
        console.log(error);                                                      // Muestra el detalle del error en consola
        return res.status(500).json({mensaje: "Error al guardar producto"})      // Responde con error al cliente
    }
}

async function mostrar(req, res){                                                // Función para buscar un producto específico por ID
    try {                                                                        // Inicia bloque de seguridad
        const {id} = req.params;                                                 // Extrae el ID enviado en la URL (ej: /producto/5)
        const producto = await Producto.findByPk(id);                            // Busca el registro por su Llave Primaria (ID)
        return res.status(200).json(producto);                                   // Retorna el producto encontrado al cliente
    } catch (error) {                                                            // Captura errores de búsqueda
        console.log(error);                                                      // Loguea el fallo
        return res.status(500).json({mensaje: "Error al mostrar el producto"})   // Informa del error al usuario
    }
}

async function modificar(req, res){                                              // Función para actualizar datos existentes
    try {                                                                        // Inicia bloque de seguridad
        const {id} = req.params;                                                 // Identifica qué producto se va a cambiar
        const producto = await Producto.findByPk(id);                            // Primero busca el producto en la base de datos
        await producto.update(req.body)                                          // Aplica los nuevos cambios enviados en la petición
        return res.status(200).json(producto);                                   // Retorna el producto ya actualizado
    } catch (error) {                                                            // Captura errores en la actualización
        return res.status(500).json({mensaje: "Error al modificar el producto"}) // Responde con error
    }
}

async function eliminar(req, res){                                               // Función para borrar un registro
    try {                                                                        // Inicia bloque de seguridad
        const { id } = req.params;                                               // Obtiene el ID del producto a eliminar
        const producto = await Producto.findByPk(id);                            // Localiza el producto en la BD
        await producto.destroy();                                                // Ejecuta el borrado físico del registro
        return res.status(200).json({mensaje: "Producto eliminado"})             // Confirma la eliminación al cliente
    } catch (error) {                                                            // Captura errores de borrado
        return res.status(500).json({mensaje: "Error al elimniar el Producto"}) // Responde con error
    }
}

module.exports = {                                                               // Exporta todas las funciones como un objeto
    listar,                                                                      // Exporta listar
    guardar,                                                                     // Exporta guardar
    mostrar,                                                                     // Exporta mostrar
    eliminar,                                                                    // Exporta eliminar
    modificar                                                                    // Exporta modificar
}
